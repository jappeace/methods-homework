\section{Prove the ILL distributivity equivalance below}
To prove $A \otimes (B \oplus C) \vdash (A \otimes B)\oplus(A\times C)$ 
and $(A \otimes B)\oplus(A\otimes C) \vdash A \otimes (B \oplus C)$


\begin{prooftree}
\RightLabel{Ax}
\AxiomC{}
\UnaryInfC{$
	(A \otimes B)\oplus(A\otimes C)
	\vdash
	A \otimes B
	\oplus
	A \otimes C
$}
\AxiomC{$
	A \otimes B
	\vdash
	A \otimes (B \oplus C)
$}
\AxiomC{$
	A \otimes C
	\vdash
	A \otimes (B \oplus C)
$}
\RightLabel{$E\oplus$}
\TrinaryInfC{$
	(A \otimes B)\oplus(A\otimes C) \vdash A \otimes (B \oplus C)
$}
\end{prooftree}
